import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class ConstantsTest {

    @Test
    void testConstantValues() {
        assertEquals(1, Constants.HIGH_PRIORITY, "HIGH_PRIORITY should be 1");
        assertEquals(5, Constants.MEDIUM_PRIORITY, "MEDIUM_PRIORITY should be 5");
        assertEquals(10, Constants.LOW_PRIORITY, "LOW_PRIORITY should be 10");
    }

    // Optional: test to call the constructor for coverage tools, though unnecessary for functionality.
    @Test
    void testConstructorCoverage() {
        // This will give coverage to the implicit constructor
        new Constants();
    }
}
