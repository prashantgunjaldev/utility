package com.ubs.cumulus.parsingService.xml.morningstar.util;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.Mockito.*;

import java.util.concurrent.locks.ReentrantLock;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;

class GlobalLockManagerTest {

    @InjectMocks
    private GlobalLockManager globalLockManager;

    @Mock
    private ReentrantLock lock;

    @BeforeEach
    void setUp() {
        MockitoAnnotations.initMocks(this);
    }

    @Test
    void testTryLock() {
        when(lock.tryLock()).thenReturn(true);
        assertTrue(globalLockManager.tryLock());

        when(lock.tryLock()).thenReturn(false);
        assertFalse(globalLockManager.tryLock());
    }

    @Test
    void testUnlock_WhenLockIsLocked() {
        when(lock.isLocked()).thenReturn(true);
        globalLockManager.unlock();
        verify(lock, times(1)).unlock();
    }

    @Test
    void testUnlock_WhenLockIsNotLocked() {
        when(lock.isLocked()).thenReturn(false);
        globalLockManager.unlock();
        verify(lock, never()).unlock();
    }
}
