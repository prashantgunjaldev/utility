// Configure Reactor Netty HTTP client with Proxy Authentication
        HttpClient httpClient = HttpClient.create()
                .proxy(proxy -> proxy
                        .type(ProxyProvider.Proxy.HTTP)
                        .address(new InetSocketAddress(proxyHost, proxyPort))
                        .username(proxyUsername)
                        .password(s -> proxyPassword));

        // Build WebClient with the custom HttpClient
        return WebClient.builder()
                .clientConnector(new reactor.netty.transport.ClientTransportConnector(httpClient))
                .build();
